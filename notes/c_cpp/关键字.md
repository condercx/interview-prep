#### C语言宏中"#“和”##"的用法
1. **（#）字符串化操作符**

    简单字符串化：

    ```c
    #define STRINGIFY(x) #x
    int main() {
        printf("%s\n", STRINGIFY(hello));      // 输出: hello
        printf("%s\n", STRINGIFY(123));        // 输出: 123
        printf("%s\n", STRINGIFY(a + b));      // 输出: a + b
        return 0;
    }
    ```

    预处理后：

    ```c
    printf("%s\n", "hello");
    printf("%s\n", "123");
    printf("%s\n", "a + b");
    ```

    实际应用场景：调试宏 - 打印变量名和值

    ```c
    #define DEBUG_PRINT(var) \
        printf(#var " = %d\n", var)

    int main() {
        int count = 42;
        int total = 100;
        
        DEBUG_PRINT(count);   // 输出: count = 42
        DEBUG_PRINT(total);   // 输出: total = 100
        
        return 0;
    }
    ```

    预处理后：

    ```c
    printf("count" " = %d\n", count);  // 字符串自动连接
    printf("total" " = %d\n", total);
    ```

2. **（##）符号连接操作符**

    用于将两个宏参数连接成一个符号

    ```c
    #define CONCAT(a, b) a##b

    int main() {
        int xy = 100;
        
        printf("%d\n", CONCAT(x, y));  // 展开为: xy
        // 输出: 100
        
        return 0;
    }
    ```

    预处理后：

    ```c
    printf("%d\n", xy);
    ```

3. **注意事项：**

    a. ## 前后的空格可有可无

    ```c
    #define CONCAT1(a, b) a##b      // 没有空格
    #define CONCAT2(a, b) a ## b    // 有空格
    #define CONCAT3(a, b) a ##b     // 左边有空格
    #define CONCAT4(a, b) a## b     // 右边有空格

    // 以上四种写法完全等价
    int xy = 10;
    printf("%d\n", CONCAT1(x, y));  // 输出: 10
    printf("%d\n", CONCAT2(x, y));  // 输出: 10
    printf("%d\n", CONCAT3(x, y));  // 输出: 10
    printf("%d\n", CONCAT4(x, y));  // 输出: 10
    ```

    b. 连接后必须是有效的标识符

    ```c
    #define CONCAT(a, b) a##b

    // ✅ 正确示例
    int var_max = 100;
    int result = CONCAT(var, _max);  // 展开为: var_max (存在的变量)

    // ❌ 错误示例
    int result = CONCAT(var, _unknown);  // 展开为: var_unknown (不存在！)
    // 编译错误: 'var_unknown' undeclared
    ```

    c. ## 会阻止宏展开（最重要！）

    基本情况:

    ```c
    #define VALUE 100
    #define CONCAT(a, b) a##b

    int VALUE_MAX = 200;

    // 期望：CONCAT(VALUE, _MAX) → 100_MAX (错误！)
    // 实际：CONCAT(VALUE, _MAX) → VALUE_MAX (正确)
    int x = CONCAT(VALUE, _MAX);  // x = 200
    ```

    示例：

    ```c
    #include <stdio.h>
    #include <string.h>
    #define STRCPY(a, b)   strcpy(a ## _p, #b)
    int main()
    {
        char var1_p[20];
        char var2_p[30];
        strcpy(var1_p, "aaaa");
        strcpy(var2_p, "bbbb");
        STRCPY(var1, var2);
        STRCPY(var2, var1);
        printf("var1 = %s\n", var1_p);
        printf("var2 = %s\n", var2_p);
        //STRCPY(STRCPY(var1,var2),var2);
        //这里是否会展开为： strcpy(strcpy(var1_p,"var2")_p,"var2“）？答案是否定的：
        //展开结果将是：  strcpy(STRCPY(var1,var2)_p,"var2")
        //## 阻止了参数的宏展开!如果宏定义里没有用到 # 和 ##, 宏将会完全展开
        // 把注释打开的话，会报错:implicit declaration of function 'STRCPY'
        return 0;
    } 
    ```

    结果：

    ```c
    var1 = var2
    var2 = var1
    ```

#### 关键字volatile有什么含意？并举出三个不同的例子？

1. **并行设备的硬件寄存器**
    存储器映射的硬件寄存器通常加volatile，因为寄存器随时可以被外设硬件修改。当声明指向设备寄存器的指针时一定要用volatile，它会告诉编译器不要对存储在这个地址的数据进行假设。

    例子：
    ```c
    #include <stdint.h>
    #include <stdio.h>

    /* ============================================
    * 模拟一个串口UART硬件寄存器
    * ============================================ */

    // 假设UART控制器的基地址
    #define UART_BASE_ADDR  0x40011000

    // UART寄存器结构体定义
    typedef struct {
        volatile uint32_t SR;    // 状态寄存器 (Status Register)
        volatile uint32_t DR;    // 数据寄存器 (Data Register)
        volatile uint32_t BRR;   // 波特率寄存器 (Baud Rate Register)
        volatile uint32_t CR1;   // 控制寄存器1 (Control Register 1)
    } UART_TypeDef;

    // 状态寄存器位定义
    #define UART_SR_TXE   (1 << 7)  // 发送数据寄存器空
    #define UART_SR_RXNE  (1 << 5)  // 接收数据寄存器非空

    // 获取UART寄存器指针
    #define UART1  ((UART_TypeDef *)UART_BASE_ADDR)

    /* ============================================
    * 示例1：发送数据函数
    * ============================================ */
    void uart_send_byte(uint8_t data) {
        // ✅ 正确：使用volatile确保每次都检查硬件状态
        // 等待发送寄存器为空
        while (!(UART1->SR & UART_SR_TXE)) {
            // 硬件会自动清除TXE标志，必须每次从内存读取
            // 如果没有volatile，编译器可能认为SR不会改变，优化成死循环
        }
        
        // 写入数据到数据寄存器
        UART1->DR = data;
    }

    /* ============================================
    * 示例2：接收数据函数
    * ============================================ */
    uint8_t uart_receive_byte(void) {
        // 等待接收寄存器非空
        while (!(UART1->SR & UART_SR_RXNE)) {
            // 硬件接收到数据后会设置RXNE标志
        }
        
        // 读取数据（读取DR会自动清除RXNE标志）
        return (uint8_t)(UART1->DR);
    }

    /* ============================================
    * 错误示例：没有使用volatile
    * ============================================ */
    typedef struct {
        uint32_t SR;    // ❌ 没有volatile
        uint32_t DR;
    } UART_Wrong_TypeDef;

    void uart_send_wrong(uint8_t data) {
        UART_Wrong_TypeDef *uart = (UART_Wrong_TypeDef *)UART_BASE_ADDR;
        
        // ❌ 编译器可能优化：
        // 1. 将SR读取到寄存器一次
        // 2. 循环中不再从内存读取
        // 3. 变成死循环，因为寄存器中的值永远不变
        while (!(uart->SR & UART_SR_TXE)) {
            // 可能被优化成死循环！
        }
        uart->DR = data;
    }

    /* ============================================
    * 测试代码
    * ============================================ */
    void test_hardware_register() {
        printf("=== 场景1：硬件寄存器访问 ===\n");
        
        // 初始化UART
        UART1->BRR = 0x683;  // 设置波特率
        UART1->CR1 = 0x200C; // 使能UART、发送、接收
        
        // 发送字符串
        const char *msg = "Hello";
        for (int i = 0; msg[i] != '\0'; i++) {
            uart_send_byte(msg[i]);
        }
        
        printf("数据发送完成\n\n");
    }
    ```

2. **一个中断服务程序中修改的供其他程序检测的变量**
    volatile提醒编译器，它后面所定义的变量随时都有可能改变。因此编译后的程序每次需要存储或读取这个变量的时候，都会直接从变量地址中读取数据。如果没有volatile关键字，则编译器可能优化读取和存储，可能暂时使用寄存器中的值，如果这个变量由别的程序更新了的话，将出现不一致的现象。

    例子：
    ```c
    #include <stdio.h>
    #include <signal.h>
    #include <unistd.h>
    #include <stdbool.h>

    /* ============================================
    * 中断标志变量
    * ============================================ */
    volatile sig_atomic_t button_pressed = 0;  // 按钮按下标志
    volatile uint32_t interrupt_count = 0;      // 中断计数器
    volatile bool system_error = false;         // 系统错误标志

    /* ============================================
    * 模拟硬件中断服务程序 (ISR)
    * ============================================ */
    void timer_interrupt_handler(int signum) {
        // 在ISR中修改volatile变量
        interrupt_count++;
        
        // 模拟每10次中断触发一次按钮事件
        if (interrupt_count % 10 == 0) {
            button_pressed = 1;
        }
        
        // 模拟错误检测
        if (interrupt_count > 100) {
            system_error = true;
        }
    }

    /* ============================================
    * 主程序检测中断标志
    * ============================================ */
    void process_button_event() {
        printf("⚡ 按钮事件处理：中断次数 = %u\n", interrupt_count);
        button_pressed = 0;  // 清除标志
    }

    void main_loop_with_volatile() {
        printf("=== 场景2：中断服务程序 (正确使用volatile) ===\n");
        
        // 设置信号处理（模拟硬件中断）
        signal(SIGALRM, timer_interrupt_handler);
        alarm(1);  // 1秒后触发
        
        int loop_count = 0;
        
        // ✅ 正确：使用volatile，编译器每次都从内存读取
        while (!system_error && loop_count < 15) {
            // 检查按钮标志
            if (button_pressed) {
                process_button_event();
            }
            
            // 模拟主程序工作
            printf("主循环运行中... (循环 %d, 中断计数 %u)\n", 
                  loop_count++, interrupt_count);
            
            sleep(1);
            alarm(1);  // 重新设置定时器
        }
        
        if (system_error) {
            printf("❌ 检测到系统错误，退出！\n");
        }
        printf("\n");
    }

    /* ============================================
    * 错误示例：不使用volatile
    * ============================================ */
    int wrong_flag = 0;  // ❌ 没有volatile

    void wrong_isr(int signum) {
        wrong_flag = 1;
    }

    void main_loop_without_volatile() {
        printf("=== 错误示例：不使用volatile ===\n");
        
        signal(SIGALRM, wrong_isr);
        alarm(1);
        
        // ❌ 编译器可能优化：
        // 1. 读取wrong_flag到寄存器（值为0）
        // 2. 认为循环内wrong_flag不会改变
        // 3. 优化成 while(1)，永远看不到ISR的修改
        
        printf("等待中断...\n");
        int timeout = 0;
        while (wrong_flag == 0 && timeout < 3) {
            printf("  检查标志... (可能看不到中断修改)\n");
            sleep(1);
            timeout++;
        }
        
        if (wrong_flag) {
            printf("✓ 检测到中断\n");
        } else {
            printf("✗ 超时：可能因为编译器优化导致看不到标志变化\n");
        }
        printf("\n");
    }

    /* ============================================
    * 实际嵌入式示例：外部中断
    * ============================================ */
    #define MAX_BUFFER_SIZE 64

    volatile uint8_t rx_buffer[MAX_BUFFER_SIZE];  // 接收缓冲区
    volatile uint32_t rx_write_index = 0;         // 写指针
    volatile uint32_t rx_read_index = 0;          // 读指针
    volatile bool rx_overflow = false;            // 溢出标志

    // 串口接收中断服务程序
    void UART_RX_IRQHandler(uint8_t received_data) {
        uint32_t next_index = (rx_write_index + 1) % MAX_BUFFER_SIZE;
        
        // 检查缓冲区是否满
        if (next_index == rx_read_index) {
            rx_overflow = true;  // 缓冲区溢出
            return;
        }
        
        // 存储接收到的数据
        rx_buffer[rx_write_index] = received_data;
        rx_write_index = next_index;
    }

    // 主程序读取数据
    bool read_uart_data(uint8_t *data) {
        // 检查是否有数据
        if (rx_read_index == rx_write_index) {
            return false;  // 缓冲区空
        }
        
        // 读取数据
        *data = rx_buffer[rx_read_index];
        rx_read_index = (rx_read_index + 1) % MAX_BUFFER_SIZE;
        
        return true;
    }

    void test_uart_interrupt() {
        printf("=== 实际应用：UART中断接收 ===\n");
        
        // 模拟接收数据
        UART_RX_IRQHandler('H');
        UART_RX_IRQHandler('i');
        UART_RX_IRQHandler('!');
        
        // 主程序读取
        uint8_t data;
        while (read_uart_data(&data)) {
            printf("接收到: '%c'\n", data);
        }
        
        if (rx_overflow) {
            printf("⚠️  警告：缓冲区溢出\n");
        }
        printf("\n");
    }
    ```

3. **多线程应用中被几个任务共享的变量**
    单地说就是防止编译器对代码进行优化.比如如下程序：
    ```c
    XBYTE[2]=0x55;
    XBYTE[2]=0x56;
    XBYTE[2]=0x57;
    XBYTE[2]=0x58;
    ```
    对外部硬件而言，上述四条语句分别表示不同的操作，会产生四种不同的动作，但是编译器却会对上述四条语句进行优化，认为只有`XBYTE[2]=0x58`（即忽略前三条语句，只产生一条机器代码）。如果键入`volatile`，编译器会逐一的进行编译并产生相应的机器代码（产生四条代码）。
    例子：
    ```c
    /* ============================================
    * 模拟外部硬件寄存器 XBYTE
    * ============================================ */
    #define XBYTE_BASE 0x8000

    // 定义外部字节数组（模拟硬件地址空间）
    #define XBYTE ((volatile uint8_t *)XBYTE_BASE)

    /* ============================================
    * 示例：LCD控制器初始化序列
    * ============================================ */
    void lcd_init_with_volatile() {
        printf("=== XBYTE示例：使用volatile ===\n");
        
        // ✅ 使用volatile，每条语句都会生成实际的写操作
        // 这些操作对硬件有不同的含义
        
        XBYTE[2] = 0x55;  // 第1步：发送命令0x55（例如：软复位）
        XBYTE[2] = 0x56;  // 第2步：发送命令0x56（例如：设置模式）
        XBYTE[2] = 0x57;  // 第3步：发送命令0x57（例如：设置对比度）
        XBYTE[2] = 0x58;  // 第4步：发送命令0x58（例如：显示开启）
        
        printf("✓ 生成了4条写操作指令\n");
        printf("  每条指令都会被执行，硬件收到完整的初始化序列\n\n");
    }

    void lcd_init_without_volatile() {
        printf("=== XBYTE示例：不使用volatile (错误) ===\n");
        
        uint8_t *xbyte_wrong = (uint8_t *)XBYTE_BASE;  // ❌ 没有volatile
        
        // ❌ 编译器优化：认为只有最后一次写入有效
        // 前三条语句可能被优化掉
        xbyte_wrong[2] = 0x55;  // 可能被优化掉
        xbyte_wrong[2] = 0x56;  // 可能被优化掉
        xbyte_wrong[2] = 0x57;  // 可能被优化掉
        xbyte_wrong[2] = 0x58;  // 只保留这一条
        
        printf("✗ 可能只生成1条写操作指令\n");
        printf("  硬件只收到最后一个值0x58，初始化失败！\n\n");
    }

    /* ============================================
    * 实际应用：步进电机控制
    * ============================================ */
    void stepper_motor_step() {
        printf("=== 步进电机控制示例 ===\n");
        
        // 步进电机需要特定的脉冲序列
        // 每一步都必须实际执行
        
        volatile uint8_t *motor_port = (volatile uint8_t *)0x9000;
        
        // 四相步进电机的步进序列
        *motor_port = 0x01;  // 相位1: 0001
        *motor_port = 0x03;  // 相位2: 0011
        *motor_port = 0x02;  // 相位3: 0010
        *motor_port = 0x06;  // 相位4: 0110
        *motor_port = 0x04;  // 相位5: 0100
        *motor_port = 0x0C;  // 相位6: 1100
        *motor_port = 0x08;  // 相位7: 1000
        *motor_port = 0x09;  // 相位8: 1001
        
        printf("✓ 完整的8步序列已发送到电机\n");
        printf("  如果没有volatile，电机可能无法正确转动\n\n");
    }

    /* ============================================
    * 汇编代码对比（伪代码）
    * ============================================ */
    void show_assembly_difference() {
        printf("=== 编译器生成的代码差异 ===\n\n");
        
        printf("使用 volatile:\n");
        printf("  MOV R0, #0x55\n");
        printf("  STR R0, [0x8002]  ; 写入0x55\n");
        printf("  MOV R0, #0x56\n");
        printf("  STR R0, [0x8002]  ; 写入0x56\n");
        printf("  MOV R0, #0x57\n");
        printf("  STR R0, [0x8002]  ; 写入0x57\n");
        printf("  MOV R0, #0x58\n");
        printf("  STR R0, [0x8002]  ; 写入0x58\n");
        printf("  → 生成4条存储指令\n\n");
        
        printf("不使用 volatile:\n");
        printf("  MOV R0, #0x58\n");
        printf("  STR R0, [0x8002]  ; 只写入0x58\n");
        printf("  → 只生成1条存储指令（前3条被优化掉）\n\n");
    }

    /* ============================================
    * 主函数
    * ============================================ */
    int main() {
        printf("╔════════════════════════════════════════╗\n");
        printf("║   volatile 关键字三大应用场景演示    ║\n");
        printf("╚════════════════════════════════════════╝\n\n");
        
        // 场景1：硬件寄存器
        test_hardware_register();
        
        // 场景2：中断服务程序
        main_loop_with_volatile();
        main_loop_without_volatile();
        test_uart_interrupt();
        
        // 场景3：多线程
        test_multithread_volatile();
        test_producer_consumer();
        
        // XBYTE示例
        lcd_init_with_volatile();
        lcd_init_without_volatile();
        stepper_motor_step();
        show_assembly_difference();
        
        printf("╔════════════════════════════════════════╗\n");
        printf("║  演示完成！记住：volatile保证可见性  ║\n");
        printf("║  但不保证原子性，需要时使用互斥锁！  ║\n");
        printf("╚════════════════════════════════════════╝\n");
        
        return 0;
    }
    ```

#### 关键字static的作用是什么？
1. 在函数体，**只会被初始化一次**，一个被声明为静态的变量在这一函数被调用过程中维持其值不变。
    例子：
    ```c
    #include <stdio.h>

    /* ============================================
    * 例子1：函数体内的static变量
    * 特点：只初始化一次，保持值不变
    * ============================================ */

    // 示例1：访客计数器
    void visitor_counter() {
        static int visitor_count = 0;  // 只在第一次调用时初始化为0
        visitor_count++;
        printf("欢迎第 %d 位访客！\n", visitor_count);
    }

    void test_example1() {
        printf("=== 例子1：函数体内的static变量 ===\n\n");
        
        printf("多次调用函数，观察static变量保持值:\n");
        visitor_counter();  // 输出: 欢迎第 1 位访客！
        visitor_counter();  // 输出: 欢迎第 2 位访客！
        visitor_counter();  // 输出: 欢迎第 3 位访客！
        visitor_counter();  // 输出: 欢迎第 4 位访客！
        visitor_counter();  // 输出: 欢迎第 5 位访客！
        
        printf("\n说明：static变量visitor_count只初始化一次，\n");
        printf("      每次调用都保持上次的值并累加。\n\n");
    }

    // 示例2：对比普通变量
    void normal_counter() {
        int count = 0;  // 每次调用都重新初始化为0
        count++;
        printf("普通变量: count = %d\n", count);
    }

    void static_counter() {
        static int count = 0;  // 只初始化一次
        count++;
        printf("static变量: count = %d\n", count);
    }

    void compare_normal_and_static() {
        printf("=== 对比：普通变量 vs static变量 ===\n\n");
        
        printf("调用普通变量函数3次:\n");
        normal_counter();  // 总是输出 1
        normal_counter();  // 总是输出 1
        normal_counter();  // 总是输出 1
        
        printf("\n调用static变量函数3次:\n");
        static_counter();  // 输出 1
        static_counter();  // 输出 2
        static_counter();  // 输出 3
        
        printf("\n");
    }

    int main() {
        test_example1();
        compare_normal_and_static();
        return 0;
    }
    ```
2. 在模块内（但在**函数体外**），一个被声明为**静态的变量**可以被模块内所用函数访问，但不能被模块外其它函数访问。它是一个**本地的全局变量**（只能被当前文件使用）。
3. 在模块内，一个被声明为**静态的函数**只可被这一模块内的其它函数调用。那就是，这个函数被限制在声明它的模块的本地范围内使用（**只能被当前文件使用**）。
4. 总结表格

    | 特性 | 普通局部变量 | static局部变量 | 全局变量 |
    |------|-------------|---------------|----------|
    | **存储位置** | 栈 | 数据段 | 数据段 |
    | **生命周期** | 函数调用期间 | 程序运行期间 | 程序运行期间 |
    | **作用域** | 函数内 | 函数内 | 全局/文件 |
    | **初始化次数** | 每次调用 | 只一次 | 只一次 |
    | **默认初始值** | 未定义 | 0 | 0 |

    | 特性 | 普通函数 | static函数 |
    |------|---------|-----------|
    | **可见性** | 全局 | 文件内 |
    | **链接属性** | 外部链接 | 内部链接 |
    | **优化潜力** | 受限 | 更好 |
    | **命名冲突** | 可能 | 不会 |

#### 在C语言中，为什么 static变量只初始化一次？

对于所有的对象（不仅仅是静态对象），**初始化都只有一次**，而由于静态变量具有“记忆”功能，初始化后，一直都没有被销毁，**都会保存在内存区域中**，所以不会再次初始化。存放在静态区的变量的生命周期一般比较长，它与整个程序“同生死、共存亡”，所以它只需初始化一次。而auto变量，即自动变量，由于它**存放在栈区**，一旦函数调用结束，**就会立刻被销毁**。

下面是一个例子： 

```c
void counter() {
    static int count = 0;  // 存储在静态存储区（数据段）
    int temp = 0;          // 存储在栈区
    
    count++;
    temp++;
    
    printf("count = %d, temp = %d\n", count, temp);
}

int main() {
    counter();  // 输出: count = 1, temp = 1
    counter();  // 输出: count = 2, temp = 1
    counter();  // 输出: count = 3, temp = 1
    return 0;
}
```

#### extern”C” 的作用是什么？

extern "C" 的主要作用是解决 C++ 与 C 语言在**函数名编译方式**上的差异，确保 C++ 代码能够正确调用 C 语言编译的函数，或者让 C 代码能够调用 C++ 编译的函数。

**C 与 C++ 编译器的差异**

```c
// 同一个函数声明
void print(int x);

// C 编译器处理后的符号名
// _print  或  print  （基本保持原样）

// C++ 编译器处理后的符号名（名称修饰）
// _Z5printi  或  ?print@@YAXH@Z  （包含参数类型信息）
```

**链接错误示例**

```c
// math.c (C 语言编译)
int add(int a, int b) {
    return a + b;
}
// 编译后符号: add

// main.cpp (C++ 编译)
int add(int a, int b);  // C++ 编译器期望找到: _Z3addii

int main() {
    int result = add(1, 2);  // 链接错误！找不到 _Z3addii
    return 0;
}
```

**场景1：C++ 调用 C 库**

```c
// 调用系统 C 库
extern "C" {
    #include <stdio.h>   // C 标准库
    #include <stdlib.h>
}

// 调用第三方 C 库
extern "C" {
    #include <sqlite3.h>  // SQLite C 库
}

int main() {
    printf("Hello\n");  // 正确链接到 C 的 printf
    sqlite3* db;
    sqlite3_open("test.db", &db);  // 正确调用 C 库函数
    return 0;
}
```

**场景2：C 调用 C++ 编译的函数**

```c
// cpp_module.h
#ifdef __cplusplus
extern "C" {
#endif

// 这些函数可以被 C 代码调用
void cpp_init();
int cpp_calculate(int x, int y);

#ifdef __cplusplus
}
#endif

// cpp_module.cpp
#include "cpp_module.h"
#include <vector>
#include <algorithm>

// 内部可以使用 C++ 特性
extern "C" void cpp_init() {
    std::vector<int> vec = {1, 2, 3};  // C++ 代码
}

extern "C" int cpp_calculate(int x, int y) {
    return std::max(x, y);  // 使用 C++ 标准库
}
```

```c
// main.c (纯 C 代码)
#include "cpp_module.h"

int main() {
    cpp_init();
    int result = cpp_calculate(10, 20);
    printf("Result: %d\n", result);
    return 0;
}
```

#### const有什么作用？

1. 定义变量（局部变量或全局变量）为常量，例如：

```c
const int N=100;//定义一个常量N
N=50; //错误，常量的值不能被修改
const int n; //错误，常量在定义的时候必须初始化
```

2. 修饰函数的参数，表示在函数体内不能修改这个参数的值。

3. 修饰函数的返回值。
    a. 如果给用 const修饰**返回值的类型为指针**，那么函数返回值（即指针）的内容是不能被修改的，而且这个返回值只能赋给被 const修饰的指针。例如：
    ```c
    const char GetString()  //定义一个函数
    char *str= GetString() //错误，因为str没有被 const修饰
    const char *str=GetString() //正确
    ```
    b. 如果用 const修饰**普通的返回值**，如返回int变量，由于这个返回值是一个临时变量，在函数调用结束后这个临时变量的生命周期也就结束了，因此把这些**返回值修饰为 const是没有意义的**。

4. 节省空间，避免不必要的内存分配。例如：

```c
#include <stdio.h>

#define MACRO_MSG "Hello World"
const char* CONST_MSG = "Hello World";

void use_macro_10_times() {
    // 每次使用宏都是文本替换
    printf("%s\n", "Hello World");  // 替换后
    printf("%s\n", "Hello World");  // 替换后
    printf("%s\n", "Hello World");  // 替换后
    // ... 重复 10 次
    
    // 编译器可能优化：合并相同的字符串字面量
    // 但这是编译器的优化，不是语言保证的
}

void use_const_10_times() {
    // 每次使用都引用同一个地址
    printf("%s\n", CONST_MSG);  // 引用 CONST_MSG
    printf("%s\n", CONST_MSG);  // 引用 CONST_MSG
    printf("%s\n", CONST_MSG);  // 引用 CONST_MSG
    // ... 重复 10 次
    
    // CONST_MSG 只有一个内存地址，语言保证
}

int main() {
    printf("CONST_MSG 地址: %p\n", (void*)&CONST_MSG);
    // printf("MACRO_MSG 地址: %p\n", (void*)&MACRO_MSG);  // ❌ 错误：宏没有地址
    
    return 0;
}
```

#### 什么情况下使用const关键字？

1. **修饰一般常量（简单类型常量）**

    `const` 可放在类型前后，含义一致。

    ```c
    const int x = 2;
    int const y = 3;
    ```

2. **修饰常数组**

    用于声明数组元素不可修改。

    ```c
    const int a[8] = {1,2,3,4,5,6,7,8};
    int const b[8] = {1,2,3,4,5,6,7,8};
    ```

3. **修饰常对象（C++）**

    常对象必须初始化，初始化后对象本身不能再被更新。

    ```cpp
    class A {};
    const A a1;
    A const a2;
    ```

4. **修饰指针（重点区分）**

    ```c
    const int *p1;        // 指向常量的指针：*p1 不可改，p1 可改
    int const *p2;        // 同上
    int *const p3 = NULL; // 常量指针：p3 不可改，*p3 可改
    const int *const p4 = NULL; // 指针和所指内容都不可改
    ```

5. **修饰引用（C++ 常引用）**

    被 `const` 修饰的引用在初始化后不能再绑定到其他对象，且不能通过该引用修改对象。

    ```cpp
    int n = 10;
    const int &r = n;
    // r = 20;  // 错误：不能通过常引用修改
    ```

6. **修饰函数形参**

    用于约束“函数内部不修改参数”，提升接口安全性，防止误改。

    ```c
    void Fun(const int var) {
        // var = 10; // 错误
    }
    ```

7. **修饰函数返回值**

    a. 若返回“指针类型”，`const` 可限制调用方不能通过返回指针改写数据。

    ```c
    const char *GetString(void);

    const char *s1 = GetString(); // 正确
    // char *s2 = GetString();     // 错误（丢弃const限定）
    ```

    b. 若返回普通值类型（如 `int`），返回的是临时值，通常再加 `const` 意义不大。

#### new/delete与malloc/free的区别是什么？

1. **所属语言不同**

    `new/delete` 是 C++ 运算符；`malloc/free` 是 C 标准库函数（`<stdlib.h>`）。

2. **返回类型与类型安全不同**

    `new` 返回“目标类型指针”，一般不需要强转；`malloc` 返回 `void*`，在 C++ 中必须显式转换。

    ```cpp
    int *p1 = new int(10);                  // 类型安全
    int *p2 = (int*)malloc(sizeof(int));    // 需要强转
    ```

3. **是否调用构造/析构函数不同（核心区别）**

    `new` 分配内存后会调用构造函数，`delete` 释放前会调用析构函数；
    `malloc/free` 只做“裸内存”分配与释放，不会触发对象生命周期函数。

    ```cpp
    class A {
    public:
        A()  { /* 构造 */ }
        ~A() { /* 析构 */ }
    };

    A *a1 = new A;                 // 调用构造
    delete a1;                     // 调用析构

    A *a2 = (A*)malloc(sizeof(A)); // 不调用构造
    free(a2);                      // 不调用析构
    ```

4. **失败处理机制不同**

    `malloc` 失败返回 `NULL`；`new` 默认失败会抛出 `std::bad_alloc`（可用 `new (std::nothrow)` 返回 `nullptr`）。

    ```cpp
    int *p3 = (int*)malloc(sizeof(int));
    if (p3 == NULL) {
        // 处理分配失败
    }

    int *p4 = new (std::nothrow) int;
    if (p4 == nullptr) {
        // 处理分配失败
    }
    ```

5. **内存大小计算方式不同**

    `new` 按类型分配，不容易写错字节数；`malloc` 需要手动写 `sizeof(...)`。

    ```cpp
    int *arr1 = new int[10];              // 自动按 int 大小计算
    int *arr2 = (int*)malloc(10 * sizeof(int));
    ```

6. **配对规则不同，不能混用**

    `new` 必须配 `delete`，`new[]` 必须配 `delete[]`；`malloc` 必须配 `free`。混用会导致未定义行为。

    ```cpp
    int *a = new int[5];
    delete[] a;   // 正确

    int *b = (int*)malloc(5 * sizeof(int));
    free(b);      // 正确

    // free(a);   // 错误：混用
    // delete b;  // 错误：混用
    ```

7. **底层关系（面试常问）**

    `new/delete` 通常最终也会调用底层内存分配器（如 `operator new` 里再调用 `malloc`），但它们在语义上比 `malloc/free` 多了一层“对象构造/析构与异常处理”。

8. **使用建议**

    C 代码里使用 `malloc/free`；C++ 代码优先使用 RAII（如 `std::vector`、`std::string`、`std::unique_ptr`），尽量少手写 `new/delete`。

#### strlen("\0") =？ sizeof("\0")=?

结论：

```c
strlen("\0") == 0
sizeof("\0") == 2
```

原因：

1. `strlen` 统计的是字符串中**终止符 `\0` 之前**的字符个数，不包含终止符本身。

    `"\0"` 这个字符串的第一个字符就是 `\0`，所以长度为 `0`。

2. `sizeof` 计算的是对象（这里是字符串字面量数组）所占的**总字节数**。

    `"\0"` 实际包含两个字符：
    a. 转义字符 `\0`（一个空字符）
    b. 编译器自动补的字符串结束符 `\0`

    所以总大小是 `2` 个字节。

代码验证：

```c
#include <stdio.h>
#include <string.h>

int main() {
    printf("strlen(\\\"\\0\\\") = %zu\n", strlen("\0"));
    printf("sizeof(\\\"\\0\\\") = %zu\n", sizeof("\0"));
    return 0;
}
```

补充（易错点）：

`'\0'` 是**字符常量**（通常是一个 `int` 值 0），`"\0"` 是**字符串字面量**（字符数组），二者不要混淆。

#### sizeof和strlen有什么区别？

`strlen` 与 `sizeof` 的差别可归纳为以下 5 点：

1. **本质不同**

    `sizeof` 是运算符（也是关键字，不是函数），`strlen` 是库函数。`sizeof(类型)` 必须带括号，`sizeof 变量` 可省略括号。

2. **返回值类型不同**

    `sizeof` 的结果类型是 `size_t`，用于表示对象大小（字节数）。

3. **参数/操作数要求不同**

    `sizeof` 可以作用于类型、变量、表达式；`strlen` 只能处理以 `\0` 结尾的字符串（`char*`/字符数组）。

4. **计算时机不同**

    `sizeof` 多数在编译期可确定（类型大小、定长数组大小）；`strlen` 一定在运行期逐字节扫描到第一个 `\0` 为止。

5. **数组退化场景表现不同**

    数组作为函数参数会退化为指针，此时在函数内 `sizeof(arr)` 得到的是指针大小，不是原数组大小；`strlen(arr)` 仍是字符串实际长度（前提是以 `\0` 结尾）。

示例：

```c
#include <stdio.h>
#include <string.h>

void foo(char arr[]) {
    printf("in foo: sizeof(arr) = %zu\n", sizeof(arr));
    printf("in foo: strlen(arr) = %zu\n", strlen(arr));
}

int main() {
    char str[20] = "0123456789";

    printf("main: sizeof(str) = %zu\n", sizeof(str));   // 20
    printf("main: strlen(str) = %zu\n", strlen(str));   // 10

    printf("strlen(\"\\0\") = %zu\n", strlen("\0"));  // 0
    printf("sizeof(\"\\0\") = %zu\n", sizeof("\0"));  // 2

    foo(str);
    return 0;
}
```

#### 不使用 sizeof，如何求int占用的字节数？

可以利用“同类型指针 `+1` 会跨过一个该类型对象”的特性来计算：

1. `&x` 是对象 `x` 的起始地址；`&x + 1` 是“下一个同类型对象”的地址（即越过一个 `int` 的位置）。
2. 把这两个地址都转成 `char*` 后相减，得到的就是字节差。

可写成宏：

```c
#define MY_SIZEOF(v) ((char*)(&(v) + 1) - (char*)(&(v)))
```

示例：

```c
#include <stdio.h>

#define MY_SIZEOF(v) ((char*)(&(v) + 1) - (char*)(&(v)))

int main(void) {
    int i = 0;
    double d = 0;

    printf("int bytes    = %td\n", MY_SIZEOF(i));
    printf("double bytes = %td\n", MY_SIZEOF(d));
    return 0;
}
```

注意：

- 该方法本质是“地址差法”，结果与平台和编译器数据模型相关（如 `int` 可能是 2/4 字节）。
- 这里是运行期计算，不是编译期常量表达式；工程代码中仍建议直接使用 `sizeof`。

#### C语言中 struct与 union的区别是什么?

`struct`（结构体）和 `union`（共用体）都可以把多个不同类型成员组织在一起，但内存模型完全不同：

1. **内存分配方式不同**

    - `struct`：各成员有各自独立存储空间。
    - `union`：所有成员共用同一段内存，起始地址相同。

2. **大小计算规则不同**

    - `struct`：总大小通常 >= 各成员大小之和，还会受对齐/填充影响。
    - `union`：总大小等于“最大成员大小”，再考虑对齐要求。

3. **成员可同时有效性不同**

    - `struct`：可同时保存并访问多个成员值。
    - `union`：同一时刻通常只应把“最后写入的成员”当作有效值；写一个成员会覆盖其他成员的字节内容。

4. **用途侧重点不同**

    - `struct`：描述一个对象的完整属性（如学生信息、设备状态等）。
    - `union`：节省内存、做多种数据格式复用、底层协议/寄存器字节视图转换（需注意可移植性）。

5. **初始化方式上的常见区别**

    - `struct`：可以按成员顺序一次初始化多个成员。
    - `union`：普通初始化通常只初始化第一个成员（或用指定初始化器明确指定成员）。

示例：

```c
#include <stdio.h>

struct S {
    int i;
    char c;
};

union U {
    int i;
    char c;
};

int main(void) {
    struct S s = {0};
    union U u = {0};

    s.i = 0x12345678;
    s.c = 'A';
    printf("struct: i = 0x%x, c = %c\n", s.i, s.c);

    u.i = 0x12345678;
    printf("union : i = 0x%x\n", u.i);
    u.c = 'A';
    printf("union : c = %c, i = 0x%x\n", u.c, u.i); // i 的字节已被覆盖

    return 0;
}
```

补充：

- 面试里常问“为什么 `union` 能省内存”：本质是成员复用同一地址空间。
- 涉及字节序、类型重解释时，`union` 代码和平台相关性较强，工程中要谨慎使用。
